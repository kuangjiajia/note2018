##### class

````javascript
//es5
function Person(name,age) {
    this.name = name
  	this.age = age
}
Person.prototype.sayName = function() {
    console.log(this.name)
}
var llp = new Person("llp",20)
llp.sayName() // "llp"

//es6
class Person {
    constructor(name,age) {
        this.name = name
      	this.age = age
    }
    sayName() {
        console.log(this.name)
    }
}
var llp = new Person("llp",20)
llp.sayName() // "llp"
````
